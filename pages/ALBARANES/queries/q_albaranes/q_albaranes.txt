SELECT
    d.id AS id_stel,
    d.document_number AS referencia,
    d.document_date AS fecha,
    COALESCE(c.name, '—') AS cliente,
    COALESCE(d.notes, d.document_number) AS titulo,
    d.total_amount AS importe_total,
    CASE d.document_state_id
        WHEN 87 THEN 'Pendiente de facturar'
        WHEN 82 THEN 'Facturado'
        WHEN 124 THEN 'Cerrado'
        WHEN 969948 THEN 'Aprobado'
        WHEN 1458062 THEN 'COMPROBADO POR GEMINI'
        WHEN 283014 THEN 'Garantia'
        WHEN 283015 THEN 'Entra en mantenimiento'
        ELSE 'Estado ' || COALESCE(d.document_state_id::text, '—')
    END AS estado,
    CASE d.document_state_id
        WHEN 87 THEN 'Gold'
        WHEN 82 THEN 'Green'
        WHEN 124 THEN 'LightBlue'
        WHEN 969948 THEN 'Purple'
        WHEN 1458062 THEN 'Orange'
        WHEN 283014 THEN 'Grey'
        WHEN 283015 THEN 'BurlyWood'
        ELSE 'Gray'
    END AS color_estado,
    CASE d.creator_id::text
        WHEN '1' THEN 'Jorge Garzón'
        WHEN '23196' THEN 'Daniel Garzón'
        WHEN '190' THEN 'Marco Garzón'
        ELSE 'Técnico ' || COALESCE(d.creator_id::text, '—')
    END AS creado_por
FROM core.documents d
LEFT JOIN core.customers c ON d.customer_id = c.id
WHERE d.document_type = 'delivery_note_work'
    AND ({{Filtro_Fecha.selectedDate}} IS NULL OR d.document_date >= {{Filtro_Fecha.selectedDate}}::date)
    AND (
        {{Filtro_Estado.selectedOptionValue}} IS NULL
        OR {{Filtro_Estado.selectedOptionValue}} = ''
        OR CASE d.document_state_id
            WHEN 87 THEN 'Pendiente de facturar'
            WHEN 82 THEN 'Facturado'
            WHEN 124 THEN 'Cerrado'
            WHEN 969948 THEN 'Aprobado'
            WHEN 1458062 THEN 'COMPROBADO POR GEMINI'
            WHEN 283014 THEN 'Garantia'
            WHEN 283015 THEN 'Entra en mantenimiento'
            ELSE 'Estado ' || COALESCE(d.document_state_id::text, '—')
        END = {{Filtro_Estado.selectedOptionValue}}
    )
    AND (
        {{Busqueda.text}} IS NULL
        OR {{Busqueda.text}} = ''
        OR d.document_number ILIKE '%' || {{Busqueda.text}} || '%'
        OR COALESCE(c.name, '') ILIKE '%' || {{Busqueda.text}} || '%'
        OR COALESCE(d.notes, '') ILIKE '%' || {{Busqueda.text}} || '%'
    )
ORDER BY d.document_date DESC
LIMIT 500;
