SELECT
  descripcion,
  trabajador,
  tipo,
  cliente,
  plataforma,
  to_char(fecha_caducidad, 'DD/MM/YYYY') AS fecha_caducidad,
  notas,
  CASE
    WHEN fecha_caducidad IS NULL THEN '⚫ Sin fecha'
    WHEN fecha_caducidad < CURRENT_DATE THEN '🔴 Caducado'
    WHEN fecha_caducidad <= CURRENT_DATE + INTERVAL '30 days' THEN '🟡 Próximo'
    ELSE '🟢 Vigente'
  END AS semaforo,
  CASE
    WHEN fecha_caducidad IS NOT NULL THEN (fecha_caducidad - CURRENT_DATE)::int
    ELSE NULL
  END AS dias_restantes
FROM garci_ops.documentacion_prl
ORDER BY
  CASE
    WHEN fecha_caducidad IS NULL THEN 3
    WHEN fecha_caducidad < CURRENT_DATE THEN 0
    WHEN fecha_caducidad <= CURRENT_DATE + INTERVAL '30 days' THEN 1
    ELSE 2
  END,
  fecha_caducidad ASC NULLS LAST
