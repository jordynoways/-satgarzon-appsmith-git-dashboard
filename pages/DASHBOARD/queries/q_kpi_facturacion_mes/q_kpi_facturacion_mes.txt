SELECT
    ROUND(COALESCE(SUM(total_amount), 0), 2) AS facturacion_mes,
    COUNT(*)::int AS num_albaranes_mes
FROM core.documents
WHERE document_type = 'delivery_note_work'
    AND document_date >= date_trunc('month', CURRENT_DATE)
    AND document_date < date_trunc('month', CURRENT_DATE) + INTERVAL '1 month';
