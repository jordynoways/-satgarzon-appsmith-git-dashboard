SELECT
    COALESCE(c.name, 'Sin cliente') AS cliente,
    COUNT(*)::int AS num_albaranes,
    ROUND(SUM(d.total_amount), 2) AS total
FROM core.documents d
LEFT JOIN core.customers c ON d.customer_id = c.id
WHERE d.document_type = 'delivery_note_work'
    AND d.document_date >= date_trunc('month', CURRENT_DATE)
GROUP BY c.name
ORDER BY total DESC
LIMIT 10;
