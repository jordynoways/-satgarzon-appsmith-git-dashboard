SELECT
    entity,
    COUNT(*) AS total_registros,
    MAX(ingested_at) AS ultima_actualizacion,
    ROUND(EXTRACT(EPOCH FROM (NOW() - MAX(ingested_at))) / 3600, 1) AS horas_desde_sync,
    CASE
        WHEN EXTRACT(EPOCH FROM (NOW() - MAX(ingested_at))) / 3600 < 2 THEN 'VERDE'
        WHEN EXTRACT(EPOCH FROM (NOW() - MAX(ingested_at))) / 3600 < 6 THEN 'AMARILLO'
        ELSE 'ROJO'
    END AS semaforo
FROM raw.stel_raw
GROUP BY entity
ORDER BY entity;
