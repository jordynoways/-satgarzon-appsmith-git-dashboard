SELECT
    date_trunc('month', document_date)::date AS mes,
    TO_CHAR(document_date, 'Mon YYYY') AS periodo,
    CASE creator_id::text
        WHEN '1' THEN 'Jorge Garzón'
        WHEN '23196' THEN 'Daniel Garzón'
        WHEN '190' THEN 'Marco Garzón'
    END AS tecnico,
    COUNT(*)::int AS num_albaranes,
    ROUND(SUM(total_amount), 2) AS facturacion
FROM core.documents
WHERE document_type = 'delivery_note_work'
    AND creator_id::text IN ('1', '23196', '190')
    AND document_date >= date_trunc('month', CURRENT_DATE) - INTERVAL '5 months'
GROUP BY mes, periodo, creator_id
ORDER BY mes DESC, facturacion DESC;
