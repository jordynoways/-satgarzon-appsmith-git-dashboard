SELECT
    year,
    month,
    TO_CHAR(TO_DATE(year::text || '-' || LPAD(month::text, 2, '0') || '-01', 'YYYY-MM-DD'), 'Mon YYYY') AS periodo,
    SUM(invoice_count)::int AS num_albaranes,
    ROUND(SUM(total_amount), 2) AS facturacion
FROM mart.customer_month_summary
WHERE (year * 12 + month) >= (EXTRACT(YEAR FROM CURRENT_DATE)::int * 12 + EXTRACT(MONTH FROM CURRENT_DATE)::int - 12)
GROUP BY year, month
ORDER BY year, month;
